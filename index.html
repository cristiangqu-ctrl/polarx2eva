    <script>

        document.addEventListener('DOMContentLoaded', () => {

            const checkboxes = document.querySelectorAll('.custom-checkbox');

            

            const updateTimeline = () => {

                checkboxes.forEach(checkbox => {

                    const taskNumber = checkbox.dataset.task;

                    const isChecked = checkbox.checked;

                    const taskCard = document.getElementById(`task-${taskNumber}`);

                    const timelineStep = document.getElementById(`step-${taskNumber}`);

                    const previousConnector = timelineStep.previousElementSibling;


                    if (isChecked) {

                        taskCard.classList.add('completed');

                        timelineStep.classList.add('completed');

                        if (previousConnector && previousConnector.classList.contains('timeline-connector')) {

                           previousConnector.classList.add('completed');

                        }

                    } else {

                        taskCard.classList.remove('completed');

                        timelineStep.classList.remove('completed');

                        if (previousConnector && previousConnector.classList.contains('timeline-connector')) {

                           previousConnector.classList.remove('completed');

                        }

                    }

                });

            };


            checkboxes.forEach(checkbox => {

                checkbox.addEventListener('change', updateTimeline);

            });


            updateTimeline(); 

        });


        function copyToClipboard(elementId) {

            const inputElement = document.getElementById(elementId);

            inputElement.select();

            inputElement.setSelectionRange(0, 99999); 


            try {

                document.execCommand('copy');

                showToast();

            } catch (err) {

                console.error('Error al copiar texto: ', err);

            }


            window.getSelection().removeAllRanges();

        }


        function showToast() {

            const toast = document.getElementById('toast');

            toast.classList.add('show');

            setTimeout(() => {

                toast.classList.remove('show');

            }, 2000);

        }
